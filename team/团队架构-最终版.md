## 🚀 快速部署脚本

### AITest01 部署（4个Agent）

```bash
#!/bin/bash
# deploy-aitest01.sh

set -e
echo "🚀 部署 AITest01 四个 Agent..."

# 创建目录
mkdir -p ~/.openclaw/workspace/agents/{小白-项目经理,小红-产品经理,小黑-测试专员,小青-运维专员}
mkdir -p ~/.openclaw/agents/{xiaobai-agent,xiaohong-agent,xiaohei-agent,xiaoqing-agent}/agent

# 创建身份文件（此处省略，见下方身份文件内容）

# 备份配置
cp ~/.openclaw/openclaw.json ~/.openclaw/openclaw.json.bak.$(date +%Y%m%d_%H%M%S)

# 添加Agent配置（使用Python脚本）
python3 << 'PYEOF'
import json
with open("/home/azureuser/.openclaw/openclaw.json", "r") as f:
    config = json.load(f)

if "agents" not in config:
    config["agents"] = {}
if "list" not in config["agents"]:
    config["agents"]["list"] = []

agents = [
    {"id": "xiaobai-agent", "name": "小白-项目经理", "workspace": "/home/azureuser/.openclaw/workspace/agents/小白-项目经理", "agentDir": "/home/azureuser/.openclaw/agents/xiaobai-agent", "identity": {"name": "小白"}},
    {"id": "xiaohong-agent", "name": "小红-产品经理", "workspace": "/home/azureuser/.openclaw/workspace/agents/小红-产品经理", "agentDir": "/home/azureuser/.openclaw/agents/xiaohong-agent", "identity": {"name": "小红"}},
    {"id": "xiaohei-agent", "name": "小黑-测试专员", "workspace": "/home/azureuser/.openclaw/workspace/agents/小黑-测试专员", "agentDir": "/home/azureuser/.openclaw/agents/xiaohei-agent", "identity": {"name": "小黑"}},
    {"id": "xiaoqing-agent", "name": "小青-运维专员", "workspace": "/home/azureuser/.openclaw/workspace/agents/小青-运维专员", "agentDir": "/home/azureuser/.openclaw/agents/xiaoqing-agent", "identity": {"name": "小青"}}
]

existing = [a.get("id") for a in config["agents"]["list"]]
for agent in agents:
    if agent["id"] not in existing:
        config["agents"]["list"].append(agent)
        print(f"✅ 添加: {agent['name']}")

with open("/home/azureuser/.openclaw/openclaw.json", "w") as f:
    json.dump(config, f, indent=2, ensure_ascii=False)
print("🎉 部署完成！")
PYEOF

# 重启
openclaw gateway restart
```
